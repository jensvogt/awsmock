    add_executable(${BINARY} ${SOURCES})

        # systemd available
        add_definitions(-DHAS_SYSTEMD)
        target_include_directories(${BINARY} PUBLIC include ../core/include ../dto/include ../db/include
                ${BSONCXX_INCLUDE_DIR} ${MONGOCXX_INCLUDE_DIR})
